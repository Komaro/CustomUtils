Required Implementation

Attribute 를 통한 빌드 에디터 구현

[AttributeUsage(AttributeTargets.Class)] public class BuilderAttribute : Attribute
[AttributeUsage(AttributeTargets.Enum)] public class BuildTypeAttribute : Attribute
[BuildType] public enum BUILD_TYPE { NONE, ANDROID, IOS }
// Builder 를 상속받는 클래스 작성 [Builder(BUILD_TYPE.IOS, BuildTarget.iOS, BuildTargetGroup.iOS)] public class Builder_IOS : Builder { protected override void OnPreProcess() { // Implement Build PreProcess } protected override void OnPostProcess() { // Implement Build ProProcess } }

Optional Implementation

[AttributeUsage(AttributeTargets.Enum)] public class BuildOptionAttribute : Attribute
[AttributeUsage(AttributeTargets.Enum)] public class DefineSymbolAttribute : Attribute
[BuildOption] public enum COMMON_BUILD_OPTION_TYPE { IL2Cpp, }
[BuildOption(BuildTargetGroup.Android)] public enum ANDROID_BUILD_OPTION_TYPE { DEBUG_ANDROID, RELEASE_ANDROID, }
[BuildOption(BuildTargetGroup.iOS)] public enum IOS_BUILD_OPTION_TYPE { DEBUG_IOS, RELEASE_IOS, }
[DefineSymbol] public enum BUILD_DEFINE_SYMBOL_TYPE { DEBUG_LOG, DISABLE_ERROR, ACTIVE_DEBUG_MODE, }
